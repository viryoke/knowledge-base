# Validated NixOS Config Templates

Complete, syntax-checked config files from a real session. Use as reference when building new configs.

## flake.nix — Multi-host, multi-arch

```nix
{
  description = "NixOS config — hardware-agnostic, GUI/TUI flexible";

  inputs = {
    nixpkgs.url = "github:nixos/nixpkgs/nixos-unstable";
    home-manager = {
      url = "github:nix-community/home-manager";
      inputs.nixpkgs.follows = "nixpkgs";
    };
    niri = {
      url = "github:sodiboo/niri-flake";
      inputs.nixpkgs.follows = "nixpkgs";
    };
    catppuccin.url = "github:catppuccin/nix";
  };

  outputs = { self, nixpkgs, home-manager, niri, catppuccin, ... }:
    let
      username = "viryoke";

      mkSystem = hostname: system: nixpkgs.lib.nixosSystem {
        inherit system;
        specialArgs = { inherit username niri; };
        modules = [
          ./hosts/${hostname}
          niri.nixosModules.niri
          catppuccin.nixosModules.catppuccin
          home-manager.nixosModules.home-manager
          {
            home-manager = {
              useGlobalPkgs = true;
              useUserPackages = true;
              extraSpecialArgs = { inherit username niri; };
              users.${username} = import ./home;
              sharedModules = [ catppuccin.homeManagerModules.catppuccin ];
            };
          }
        ];
      };
    in
    {
      nixosConfigurations.desktop = mkSystem "desktop" "x86_64-linux";
      nixosConfigurations.nixos   = mkSystem "nixos"   "aarch64-linux";
    };
}
```

## hosts/<hostname>/default.nix — Conditional GPU + GUI

Key pattern: define `options.myConfig` then use `lib.mkIf config.myConfig.*` everywhere.

```nix
{ config, pkgs, lib, username, ... }:
{
  imports = [ ./hardware-configuration.nix ];

  options.myConfig = {
    hasNvidia = lib.mkOption {
      type = lib.types.bool;
      default = false;
      description = "Whether this machine has an NVIDIA GPU";
    };
    hasGUI = lib.mkOption {
      type = lib.types.bool;
      default = true;
      description = "Whether to enable desktop environment";
    };
  };

  home-manager.extraSpecialArgs = {
    hasGUI = config.myConfig.hasGUI;
  };

  boot.loader.systemd-boot.enable = true;
  boot.loader.efi.canTouchEfiVariables = true;
  boot.kernelPackages = pkgs.linuxPackages_lts;

  # GPU — merged into single services.xserver block (PITFALL: no duplicates!)
  services.xserver = {
    videoDrivers = lib.mkIf config.myConfig.hasNvidia [ "nvidia" ];
    enable = lib.mkIf config.myConfig.hasGUI true;
    desktopManager.gnome.enable = lib.mkIf config.myConfig.hasGUI true;
  };

  hardware.nvidia = lib.mkIf config.myConfig.hasNvidia {
    modesetting.enable = true;
    open = false;
    nvidiaSettings = true;
    powerManagement.enable = true;
    package = config.boot.kernelPackages.nvidiaPackages.stable;
  };

  hardware.graphics = { enable = true; enable32Bit = true; };

  environment.sessionVariables = lib.mkIf config.myConfig.hasNvidia {
    LIBVA_DRIVER_NAME = "nvidia";
    __GLX_VENDOR_LIBRARY_NAME = "nvidia";
    GBM_BACKEND = "nvidia-drm";
  };

  boot.kernelParams = lib.mkIf config.myConfig.hasNvidia [ "nvidia-drm.fbdev=1" ];
  hardware.nvidia-container-toolkit.enable = config.myConfig.hasNvidia;

  # GUI — all wrapped in lib.mkIf hasGUI
  programs.niri.enable = config.myConfig.hasGUI;
  services.greetd = lib.mkIf config.myConfig.hasGUI { /* ... */ };
  xdg.portal = lib.mkIf config.myConfig.hasGUI { /* ... */ };

  # Always-on
  zramSwap = { enable = true; algorithm = "zstd"; memoryPercent = 25; };
  # ... pipewire, ssh, fonts, user, docker, nix settings ...
}
```

## hosts/<hostname>/hardware-configuration.nix — Per-machine markers

```nix
{ config, lib, pkgs, modulesPath, ... }:
{
  imports = [ (modulesPath + "/installer/scan/not-detected.nix") ];

  # ⭐ Only two lines to set per machine
  myConfig.hasNvidia = true;
  myConfig.hasGUI = true;

  # Rest is auto-generated by nixos-generate-config
  boot.initrd.availableKernelModules = [ "nvme" "xhci_pci" ];
  fileSystems."/" = { device = "/dev/disk/by-label/nixos"; fsType = "ext4"; };
  fileSystems."/boot" = { device = "/dev/disk/by-label/boot"; fsType = "vfat"; };
  swapDevices = [ ];
  nixpkgs.hostPlatform = lib.mkDefault "x86_64-linux";
  hardware.enableRedistributableFirmware = true;
}
```

## home/default.nix — Conditional GUI packages

```nix
{ pkgs, lib, username, hasGUI, ... }:
{
  imports =
    (lib.optionals hasGUI [ ./niri.nix ])
    ++ [ ./dev.nix ];

  home.username = username;
  home.homeDirectory = "/home/${username}";
  home.stateVersion = "25.05";

  # Conditional packages
  home.packages = with pkgs; [
    ripgrep fd eza zoxide jq tree htop btop tldr zellij
    python312 uv nodejs_22 bun unzip p7zip
  ] ++ (lib.optionals hasGUI [
    vscode obsidian google-chrome firefox telegram-desktop
    mpv imv zathura wl-clipboard grim slurp satty
  ]);

  # Conditional programs
  programs.ghostty = lib.mkIf hasGUI { enable = true; };
  programs.yazi = lib.mkIf hasGUI { enable = true; };

  # Conditional session variables
  home.sessionVariables = {
    EDITOR = "nvim";
  } // (lib.optionalAttrs hasGUI {
    NIXOS_OZONE_WL = "1";
    BROWSER = "google-chrome";
    TERMINAL = "ghostty";
  });

  # Hermes Agent install (official script)
  home.activation.installHermes = lib.hm.dag.entryAfter [ "writeBoundary" ] ''
    if ! command -v hermes &>/dev/null; then
      $DRY_RUN_CMD curl -fsSL https://raw.githubusercontent.com/NousResearch/hermes-agent/main/scripts/install.sh | bash 2>&1 || echo "Hermes install failed"
    fi
  '';
}
```
